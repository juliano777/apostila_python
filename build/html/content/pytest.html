33. pytest¶

$ vim /tmp/test_pytest.py

#=============================================================================

# Função fatorial def fatorial(x):

if x == 0:

return 1

return x * fatorial(x - 1)

# Função que testa se um número é par def e_par(x):

if x % 2 == 0:

return True

return False

# Função para testar a função fatorial def test_fatorial():

assert fatorial(0) == 1

# Função para testar a função e_par def test_par():

assert e_par(8)

#=============================================================================
